    function formatLog() {
      const input = document.getElementById('input').value;
      const lines = input.split(/\n+/);
      const output = lines.map(line => {
        if (line.startsWith("Britny said:")) {
          return `<p><span class='speaker'>Britny:</span> ${line.replace("Britny said:", '').trim()}</p>`;
        } else if (line.startsWith("Nova said:")) {
          return `<p><span class='speaker'>Nova:</span> ${line.replace("Nova said:", '').trim()}</p>`;
        } else if (line.startsWith("Sherwood said:")) {
          return `<p><span class='speaker'>Sherwood:</span> ${line.replace("Sherwood said:", '').trim()}</p>`;
        } else {
          return `<p>${line}</p>`;
        }
      }).join("\n");

      document.getElementById('output').innerHTML = output;
    }

    function downloadLog() {
      const content = document.getElementById('output').innerHTML;
      const fullHTML = `<!DOCTYPE html><html lang='en'><head><meta charset='UTF-8'><title>Vault Log</title><style>body{font-family:'JetBrains Mono',monospace;background:#0e0b1a;color:#e3d1ff;padding:2rem;line-height:1.6;}.speaker{font-size:1.6em;font-weight:bold;color:#FFD5FA;text-shadow:0 0 6px #a48af9;}</style></head><body>${content}</body></html>`;
      const blob = new Blob([fullHTML], { type: 'text/html' });
      const link = document.createElement('a');
      link.href = URL.createObjectURL(blob);
      link.download = 'vault-log.html';
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
    }
  </script>
